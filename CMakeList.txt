cmake_minimum_required(VERSION 3.16)
project(sim LANGUAGES CXX)


set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

find_package(Geant4 REQUIRED ui_all vis_all)

file(GLOB sources ${PROJECT_SOURCE_DIR}/src/*.cc)
file(GLOB headers ${PROJECT_SOURCE_DIR}/include/*.hh)

add_executable(examplesim1 examplesim1.cc ${sources} ${headers})
target_include_directories(examplesim1 PRIVATE ${PROJECT_SOURCE_DIR}/include)
target_link_libraries(examplesim1 PRIVATE ${Geant4_LIBRARIES})



# Copy macros to build directory
file(GLOB MAC_FILES ${PROJECT_SOURCE_DIR}/*.mac)
foreach(f ${MAC_FILES})
  configure_file(${f} ${CMAKE_CURRENT_BINARY_DIR}/ COPYONLY)
endforeach()
